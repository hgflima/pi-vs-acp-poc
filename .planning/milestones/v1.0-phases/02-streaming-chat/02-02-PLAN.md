    ```typescript
    import type { SSEEvent } from "./types";

    export async function* parseSSEStream(
      response: Response,
      signal?: AbortSignal
    ): AsyncGenerator<SSEEvent> {
      if (!response.body) throw new Error("No response body");

      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let buffer = "";
      let currentEvent = "";
      let currentData = "";

      try {
        while (true) {
          if (signal?.aborted) break;
          const { done, value } = await reader.read();
          if (done) break;

          buffer += decoder.decode(value, { stream: true });
          const lines = buffer.split("\n");
          buffer = lines.pop() || "";

          for (const line of lines) {
            if (line.startsWith("event: ")) {
              currentEvent = line.slice(7).trim();
            } else if (line.startsWith("data: ")) {
              currentData += line.slice(6);
            } else if (line === "") {
              if (currentEvent && currentData) {
                try {
                  const parsed = JSON.parse(currentData);
                  yield { type: currentEvent, ...parsed } as SSEEvent;
                } catch {
                  // Skip malformed events
                }
              }
              currentEvent = "";
              currentData = "";
            }
          }
        }
      } finally {
        reader.releaseLock();
      }
    }
    ```

    Key details:
    - Import `SSEEvent` type from `./types` (already defined in Phase 1)
    - Uses `AsyncGenerator<SSEEvent>` return type
    - Handles multi-line data fields by concatenating (data line appends to currentData)
    - Empty line triggers event emission per SSE spec
    - `signal?.aborted` check enables AbortController integration
    - `reader.releaseLock()` in finally block ensures cleanup
    - Malformed JSON is silently skipped (try/catch around JSON.parse)

    **ChatAction union type:**
    ```typescript
    type ChatAction =
      | { type: "ADD_USER_MESSAGE"; content: string }
      | { type: "START_STREAMING" }
      | { type: "APPEND_TEXT_DELTA"; data: string }
      | { type: "STOP_STREAMING" }
      | { type: "SET_ERROR"; message: string }
      | { type: "CLEAR_ERROR" }
      | { type: "CLEAR_MESSAGES" };
    ```

    **chatReducer function:**
    - `ADD_USER_MESSAGE`: Append a `UserMessage` with `{ role: "user", content: action.content, timestamp: Date.now() }` to messages array
    - `START_STREAMING`: Set `streaming: true`, `error: null`, append a new `AssistantMessage` with `{ role: "assistant", segments: [], streaming: true, timestamp: Date.now() }`
    - `APPEND_TEXT_DELTA`: Clone the messages array. Get the last message (cast to AssistantMessage). Clone its segments array. If the last segment is a TextSegment, create a new segment with `content: lastSeg.content + action.data`. Otherwise push a new `{ type: "text", content: action.data }` TextSegment. Replace the last message with new segments. **CRITICAL: Create new object references at every level (Pitfall 6) — spread operators for messages array, last message object, segments array, and last segment object.**
    - `STOP_STREAMING`: Clone messages, find last message, if it's an assistant message set `streaming: false` on it. Set state `streaming: false`.
    - `SET_ERROR`: Set `error: action.message`. Also dispatch STOP_STREAMING behavior (set streaming false, mark last assistant message as not streaming).
    - `CLEAR_ERROR`: Set `error: null`
    - `CLEAR_MESSAGES`: Reset to initial state `{ messages: [], streaming: false, error: null }`

    **useChat hook:**
    ```typescript
    export function useChat() {
      const [state, dispatch] = useReducer(chatReducer, {
        messages: [],
        streaming: false,
        error: null,
      });
      const abortControllerRef = useRef<AbortController | null>(null);

      const sendMessage = useCallback(async (content: string, config: { model: string; provider: "anthropic" | "openai" }) => {
        dispatch({ type: "ADD_USER_MESSAGE", content });
        dispatch({ type: "START_STREAMING" });

        const controller = new AbortController();
        abortControllerRef.current = controller;

        try {
          const response = await fetch("/api/chat", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
              message: content,
              model: config.model,
              provider: config.provider,
            }),
            signal: controller.signal,
          });

          if (!response.ok) {
            let errMsg = "Request failed";
            try {
              const err = await response.json();
              errMsg = err.error || errMsg;
            } catch { /* use default */ }
            dispatch({ type: "SET_ERROR", message: errMsg });
            return;
          }

          for await (const event of parseSSEStream(response, controller.signal)) {
            switch (event.type) {
              case "text_delta":
                dispatch({ type: "APPEND_TEXT_DELTA", data: event.data });
                break;
              case "error":
                dispatch({ type: "SET_ERROR", message: event.message });
                break;
              case "done":
                break;
            }
          }
        } catch (err) {
          if (err instanceof DOMException && err.name === "AbortError") {
            // User clicked stop -- not an error, text is preserved
          } else {
            dispatch({
              type: "SET_ERROR",
              message: err instanceof Error ? err.message : "Connection error",
            });
          }
        } finally {
          dispatch({ type: "STOP_STREAMING" });
          abortControllerRef.current = null;
        }
      }, []);

      const stopGeneration = useCallback(() => {
        abortControllerRef.current?.abort();
      }, []);

      const clearMessages = useCallback(() => {
        abortControllerRef.current?.abort();
        dispatch({ type: "CLEAR_MESSAGES" });
      }, []);

      const clearError = useCallback(() => {
        dispatch({ type: "CLEAR_ERROR" });
      }, []);

      return {
        messages: state.messages,
        streaming: state.streaming,
        error: state.error,
        sendMessage,
        stopGeneration,
        clearMessages,
        clearError,
      };
    }
    ```

    - AbortError is explicitly caught and treated as non-error (user-initiated stop)
    - Non-200 responses are handled before entering the SSE parse loop
    - `clearMessages` also aborts any in-progress stream
